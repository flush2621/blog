#include<Windows.h>

//回调函数
LRESULT CALLBACK WindowProc(HWND hwnd, UINT uMsg, WPARAM wParam, LPARAM lParam);
//鼠标钩子处理函数
LRESULT CALLBACK MouseProc(int nCode, WPARAM wParam, LPARAM lParam);

//WinMain:相当于main函数，入口函数，窗口应用
//HINSTANCE:应用程序实例句柄类型
//LPSTR: char*

HHOOK g_hMouseHook;
HHOOK g_hKeyboardHook;
HWND g_hwnd;

int WINAPI WinMain(HINSTANCE hInstance, HINSTANCE hPrevInstance, LPSTR lpCmdLine, int nShowCmd)
{
	//1、创建窗口类
	WNDCLASS wc;
	wc.cbClsExtra = 0;	//窗口类的额外空间大小
	wc.cbWndExtra = 0;
	wc.hbrBackground = (HBRUSH)GetStockObject(WHITE_BRUSH); //白色画刷
	wc.hCursor = LoadCursor(NULL, IDC_ARROW); //光标句柄
	wc.hIcon = LoadIcon(NULL, IDI_APPLICATION);//图标句柄
	wc.hInstance = hInstance;	//当前应用程序实例句柄
	wc.lpfnWndProc = WindowProc; //窗口处理函数
	wc.lpszClassName = TEXT("MSGWindow");  //窗口类型名
	wc.lpszMenuName = NULL;	
	wc.style = 0;	//窗口类的风格
	//2、注册窗口类
	RegisterClass(&wc);
	//3、创建窗口
	//HWND hwnd = CreateWindow(wc.lpszClassName, TEXT("Hello_World"), WS_OVERLAPPEDWINDOW, CW_USEDEFAULT, CW_USEDEFAULT, CW_USEDEFAULT, CW_USEDEFAULT, NULL, NULL, hInstance, NULL);
	HWND hwnd = CreateWindow(wc.lpszClassName, TEXT("Hello_Milo"), WS_OVERLAPPEDWINDOW, 400, 200, 800, 600, NULL, NULL, hInstance, NULL);
	g_hwnd = hwnd;
	//4、显示窗口
	ShowWindow(hwnd, SW_SHOW);
	//5、更新窗口
	UpdateWindow(hwnd);
	MSG msg;
	//6、消息循环
	while (GetMessage(&msg, NULL, 0, 0))
	{
		//将虚拟键消息转换为字符消息WM_CHAR
		TranslateMessage(&msg);
		//将消息分发给窗口处理函数
		DispatchMessage(&msg);
	}
	return 0;
}

//1.进程内钩子HOOK：只对当前进程起作用
//2.全局钩子：针对系统所有进程，需要一个DLL，既注入技术
LRESULT CALLBACK WindowProc(HWND hwnd, UINT uMsg, WPARAM wParam, LPARAM lParam)
{
	switch (uMsg)
	{
	case WM_CREATE:	//窗口创建消息
		g_hMouseHook = SetWindowsHookEx(WH_MOUSE,MouseProc,NULL,GetCurrentThreadId());
		g_hKeyboardHook = SetWindowsHookEx(WH_KEYBOARD, KeyboardProc, NULL, GetCurrentThreadId());
		break;
	case WM_MOUSEMOVE:
		{
			int x = LOWORD(lParam);
			int y = HIWORD(lParam);
			TCHAR str[255];
			wsprintf(str, L"当前鼠标坐标(%d,%d)", x, y);
			SetWindowText(hwnd, str);
		}
		break;
	case WM_LBUTTONDOWN:
		MessageBox(hwnd, L"鼠标按下", L"提示", MB_OK);
		break;
	case WM_KEYDOWN:
		MessageBox(hwnd, L"键盘按下", L"提示", MB_OK);
		break;
	case WM_CLOSE:		//窗口关闭消息
		if (MessageBox(hwnd, L"Really quit?", L"My application", MB_OKCANCEL) == IDOK)
		{
			DestroyWindow(hwnd);
		}
		return 0;
	case WM_DESTROY:	//窗口销毁消息
		PostQuitMessage(0);//WM_QUIT
		break;
	}
	return DefWindowProc(hwnd, uMsg, wParam, lParam);
}

#include "mfc.h"

//有且仅有一个全局应用程序类对象
MyApp myApp;

//程序的入口地址
BOOL MyApp::InitInstance()
{
	/*
	1.创建框架类对象
	2.显示窗口
	3.更新窗口
	4.保存框架对象指针
	*/
	MyFrame* frame = new MyFrame;//1
	frame->ShowWindow(SW_SHOWNORMAL);//2
	frame->UpdateWindow();//3
	m_pMainWnd = frame;//4
	return TRUE;
}

MyFrame::MyFrame()
{
	//CWnd类的成员函数 CWnd::Create
	//CFrameWnd继承于CWnd
	Create(NULL, TEXT("MFC"));
}
