# 二、检查网络抛出有效数组

get_ping

#! /usr/bin/perl
$pw="headof/feitian";
$city=$ARGV[0];

use DBI;
$dbh=DBI->connect('dbi:Oracle:',"$pw",'');

$sql_a="select old_store_no
        from store_compare
        where status=1
        and area_no=?";
$sth_a=$dbh->prepare($sql_a);
$sth_a->execute($city);
while ($sno = $sth_a->fetchrow_array) {
        printf ("%-2d ",$sno);
}
print "\n";
$sth_a->finish;

$dbh->disconnect;
